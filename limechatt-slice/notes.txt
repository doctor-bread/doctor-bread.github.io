function simpleHash(input) {
  let hash = 0;
  for (let i = 0; i < input.length; i++) {
    const char = input.charCodeAt(i);
    hash = (hash << 5) - hash + char;
    hash |= 0;
  }

  const numericHash = Math.abs(hash).toString().slice(0, 8);

  return `${numericHash}`;
}

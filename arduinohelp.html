// If you are using an MCP23017, use the line below:

void setup() {
  // Initialize the LCD (16 columns, 2 rows)
  lcd.begin(16, 2);
  
  // Turn on the backlight (if supported by your backpack)
  lcd.setBacklight(HIGH);
  
  lcd.setCursor(0, 0);
  lcd.print("MCP Expander");

  lcd.print("Status: Online");
}
